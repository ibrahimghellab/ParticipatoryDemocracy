FROM Internaute I
JOIN Membre M ON I.idInternaute = M.idInternaute

JOIN Groupe G ON M.idGroupe = G.idGroupe;

FROM Proposition P
JOIN Membre M ON P.idMembre = M.idMembre
JOIN Internaute I ON M.idInternaute = I.idInternaute
JOIN Vote V ON P.idVote = V.idVote;

SELECT C.texte, P.titre, I.nom, I.prenom 
FROM Commentaire C
JOIN Proposition P ON C.idProposition = P.idProposition
JOIN ajoute A ON C.idCommentaire = A.idCommentaire
JOIN Membre M ON A.idMembre = M.idMembre
JOIN Internaute I ON M.idInternaute = I.idInternaute;

SELECT B.themeDuBudget, SUM(B.budgetGlobal) AS totalBudget 
FROM Budget B
GROUP BY B.themeDuBudget;

FROM Proposition P
JOIN est_reagit E ON P.idProposition = E.idProposition
JOIN Reaction R ON E.idReaction = R.idReaction
GROUP BY P.titre, R.emoticone;

CREATE TABLE Membre(
   idMembre INT,
   dateAdhesion VARCHAR(50),
   status VARCHAR(50),
   idInternaute INT NOT NULL,
   idRole VARCHAR(50) NOT NULL,
   idGroupe INT NOT NULL,
   PRIMARY KEY(idMembre),

);

CREATE TABLE Proposition(
   idProposition INT,
   titre VARCHAR(50),
   description VARCHAR(50),
   dateCreation DATE,
   theme VARCHAR(50),
   status VARCHAR(50),
   voteDemande LOGICAL,
   idVote INT NOT NULL,
   idMembre INT NOT NULL,
   PRIMARY KEY(idProposition),
   UNIQUE(idVote),
